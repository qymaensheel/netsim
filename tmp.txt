template<class Node> void NodeCollection<Node>::remove_by_id(ElementID id_to_rem) {
    auto it_to_rem=find_by_id(id_to_rem);
    if(it_to_rem!=NodesVector.end()){
        NodesVector.erase(find_by_id(id_to_rem));
    }
    //(it_to_rem!=NodesVector.end())? NodesVector.erase(find_by_id(id_to_rem)):std::cout<<"id ordered to be removed was not found";
}

void Factory::do_work(Time t) {
    for(auto& worker: Workers){
        worker.do_work(t);
    }
}
